```tsx
'use client';
import { useState, useEffect } from 'react';
import { createBrowserClient } from '@supabase/ssr';
import { useRouter } from 'next/navigation';
import { ALL_COLLEGES } from '@/lib/collegeList';
import { CITIES } from '@/lib/types';

export default function OnboardingPage() {
  const router = useRouter();
  const supabase = createBrowserClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!
  );
  const [selectedCity, setSelectedCity] = useState('Philadelphia');
  const [selectedCollege, setSelectedCollege] = useState('');
  const [saving, setSaving] = useState(false);
  const [error, setError] = useState('');

  // Auto-detect city from browser geolocation (best-effort, non-blocking)
  useEffect(() => {
    if (!navigator.geolocation) return;
    navigator.geolocation.getCurrentPosition(
      (pos) => {
        // Simple nearest-city heuristic using CITIES array lat/lng
        // Fall back to Philadelphia if detection fails
      },
      () => { /* ignore permission denied */ },
      { timeout: 3000 }
    );
  }, []);

  const colleges = ALL_COLLEGES.filter(c => c.city === selectedCity);

  const handleSave = async () => {
    setSaving(true);
    setError('');
    const { data: { user } } = await supabase.auth.getUser();
    if (!user) { router.push('/login'); return; }

    const { error: updateError } = await supabase
      .from('profiles')
      .update({
        university_name: selectedCollege || null,
        city: selectedCity,
      })
      .eq('id', user.id);

    if (updateError) {
      setError('Failed to save. Continuing anyway...');
      setTimeout(() => router.push('/'), 1500);
    } else {
      router.push('/');
    }
    setSaving(false);
  };

  return (
    // Dark bg matching login page: bg-[#050913], text-slate-200
    // City select using CITIES array
    // College select showing colleges filtered by city (scrollable list, max-h-48)
    // "Skip for now" button that calls handleSave() with no college selected
    // "Get Started" button (disabled while saving)
    // Error message shown in red if save fails
  );
}
```

Use the existing Tailwind dark theme (bg-[#050913], border-slate-700, indigo-500 accents) to match the login page. The city selector must use CITIES from @/lib/types (same source as login page). The college list must use ALL_COLLEGES from @/lib/collegeList filtered by selected city. The save MUST work even if the profiles row doesn't exist yet (use upsert with { id: user.id, ... } instead of update, to handle edge cases).
